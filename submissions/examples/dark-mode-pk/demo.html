Proposed Implementation

core/variables.css
:root {
  --ease-bg-primary: #ffffff;
  --ease-bg-secondary: #f8fafc;
  --ease-text-primary: #1e293b;
  --ease-text-secondary: #64748b;
  --ease-border: #e2e8f0;
  --ease-shadow: rgba(0, 0, 0, 0.08);
}

@media (prefers-color-scheme: dark) {
  :root {
    --ease-bg-primary: #1e293b;
    --ease-bg-secondary: #0f172a;
    --ease-text-primary: #f1f5f9;
    --ease-text-secondary: #94a3b8;
    --ease-border: #334155;
    --ease-shadow: rgba(0, 0, 0, 0.4);
  }
}

.ease-dark {
  --ease-bg-primary: #1e293b;
  --ease-bg-secondary: #0f172a;
  --ease-text-primary: #f1f5f9;
  --ease-border: #334155;
}

.ease-light {
  --ease-bg-primary: #ffffff;
  --ease-bg-secondary: #f8fafc;
  --ease-text-primary: #1e293b;
  --ease-border: #e2e8f0;
}
Component usage (cards.css)
.ease-card {
  background: var(--ease-bg-primary);
  color: var(--ease-text-primary);
  border: 1px solid var(--ease-border);
  box-shadow: 0 1px 3px var(--ease-shadow);
  transition: background 0.3s, color 0.3s, border 0.3s;
}
