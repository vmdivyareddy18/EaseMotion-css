How to Use

html
<div class="acd-search-wrap">
  <div class="acd-input-row">
    <input class="acd-input" type="text" />
  </div>

  <!-- Hidden by default; add .ease-autocomplete-dropdown--open to show -->
  <ul class="ease-autocomplete-dropdown">
    <li class="acd-suggestion" style="--i:0">Suggestion 1</li>
    <li class="acd-suggestion" style="--i:1">Suggestion 2</li>
    <li class="acd-suggestion" style="--i:2">Suggestion 3</li>
  </ul>
</div>
css — the core animation
/* Dropdown container: scaleY(0) → scaleY(1) from top */
.ease-autocomplete-dropdown {
  transform-origin: top;
  transform: scaleY(0);
  opacity: 0;
  transition: transform 0.22s cubic-bezier(0.34, 1.56, 0.64, 1),
              opacity 0.18s ease;
}

.ease-autocomplete-dropdown--open {
  transform: scaleY(1);
  opacity: 1;
}

/* Individual suggestions stagger via --i custom property */
.acd-suggestion {
  opacity: 0;
  transform: translateY(-6px);
  animation: none;
}

.ease-autocomplete-dropdown--open .acd-suggestion {
  animation: ease-kf-suggestion-in 0.25s ease forwards;
  animation-delay: calc(0.05s + var(--i, 0) * 0.04s);
}

@keyframes ease-kf-suggestion-in {
  to { opacity: 1; transform: translateY(0); }
}
