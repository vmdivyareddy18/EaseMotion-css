src/extension.ts
import * as vscode from 'vscode';
import { EaseMotionCompletionProvider } from './completions';
import { EaseMotionHoverProvider } from './hover';

export function activate(context: vscode.ExtensionContext) {
  const animationData = loadAnimationData();

  const completionProvider = new EaseMotionCompletionProvider(animationData);
  const hoverProvider = new EaseMotionHoverProvider(animationData);

  context.subscriptions.push(
    vscode.languages.registerCompletionItemProvider(
      ['html', 'javascript', 'typescript', 'vue', 'svelte', 'css'],
      completionProvider,
      '.', '-'
    ),
    vscode.languages.registerHoverProvider(
      ['html', 'javascript', 'typescript', 'vue', 'svelte', 'css'],
      hoverProvider
    ),
    vscode.commands.registerCommand(
      'easemotion.insertAnimation',
      openAnimationPicker
    )
  );
}
src/completions.ts
import * as vscode from 'vscode';

export class EaseMotionCompletionProvider
  implements vscode.CompletionItemProvider {

  constructor(private data: AnimationData[]) {}

  provideCompletionItems(
    document: vscode.TextDocument,
    position: vscode.Position,
  ): vscode.CompletionItem[] {
    const linePrefix = document.lineAt(position).text
      .substring(0, position.character);

    if (!linePrefix.includes('ease-')) return [];

    return this.data.map(anim => {
      const item = new vscode.CompletionItem(
        anim.className,
        vscode.CompletionItemKind.Keyword
      );
      item.detail = anim.description;
      item.documentation = new vscode.MarkdownString(
        anim.getDocs()
      );
      return item;
    });
  }
}

const fs = require('fs');
const glob = require('glob');

const data = [];
const classRegex = /\.(ease-[\w-]+)\s*\{/g;
const keyframeRegex = /@keyframes\s+(ease-[\w-]+)/g;

for (const file of glob.sync('core/**/*.css')) {
  const content = fs.readFileSync(file, 'utf8');
  // Extract class names
  for (const match of content.matchAll(classRegex)) {
    data.push({
      className: match[1],
      file,
      type: 'class',
      description: extractComment(content, match.index),
    });
  }
}

fs.writeFileSync(

  JSON.stringify(data, null, 2)
