Implementation

The fixed marquee uses a lightweight JS enhancement (shown below). Paste this into any page to fix both bugs without changing your marquee markup.

// Fix #1: Pause marquee during resize
let resizeTimer;
window.addEventListener('resize', () => {
  document.querySelectorAll('.ease-marquee-track')
    .forEach(el => el.style.animationPlayState = 'paused');
  clearTimeout(resizeTimer);
  resizeTimer = setTimeout(() => {
    document.querySelectorAll('.ease-marquee-track')
      .forEach(el => el.style.animationPlayState = '');
  }, 150);
});

// Fix #2: Pause marquee when tab is hidden
document.addEventListener('visibilitychange', () => {
  document.querySelectorAll('.ease-marquee-track')
    .forEach(el => {
      el.style.animationPlayState =
        document.hidden ? 'paused' : 'running';
    });
});
