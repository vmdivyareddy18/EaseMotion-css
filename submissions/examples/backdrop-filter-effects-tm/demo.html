Usage Example

Copy the CSS classes below to use in your projects:

<!-- Add to your HTML head -->
<link rel="stylesheet" href="submissions/examples/backdrop-filter-effects-tm/style.css">

<!-- Apply the utility class -->
<!-- Frosted glass card -->
<div class="frosted-glass">
  <h3>Glass Card Content</h3>
  <p>Semi-transparent with blur effect.</p>
</div>

<!-- Brightness overlay -->
<div class="brightness-overlay-dim">Dimmed content area</div>
