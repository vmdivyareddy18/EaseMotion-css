Usage Example

Copy the CSS classes below to use in your projects:

<!-- Add to your HTML head -->
<link rel="stylesheet" href="submissions/examples/typing-text-animations-tm/style.css">

<!-- Apply the utility class -->
<!-- Typewriter text -->
<span class="typewriter-md">Typewriter effect text...</span>

<!-- Text reveal animation -->
<div class="text-reveal-lg">Revealed on load</div>

<!-- Wave reveal -->
<span class="text-reveal-wave-md">Wave animation text</span>
