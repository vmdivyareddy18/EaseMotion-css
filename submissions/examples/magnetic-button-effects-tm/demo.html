Usage Example

Copy the CSS classes below to use in your projects:

<!-- Add to your HTML head -->
<link rel="stylesheet" href="submissions/examples/magnetic-button-effects-tm/style.css">

<!-- Apply the utility class -->
<!-- Magnetic pull button -->
<button class="magnetic-pull magnetic-primary magnetic-pull-md">Click Me</button>

<!-- Magnetic border glow -->
<div class="magnetic-border magnetic-outline magnetic-border-lg">Hover Me</div>

<!-- Magnetic scale with bounce -->
<a class="magnetic-scale magnetic-secondary magnetic-scale-sm" href="#">Link</a>
