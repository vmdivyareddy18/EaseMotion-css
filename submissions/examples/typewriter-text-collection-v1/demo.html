Quick Start

<!-- 1. Link the stylesheet -->
<link rel="stylesheet" href="style.css">

<!-- 2. Add class + char-count helper -->
<h1 class="ease-typewriter ease-chars-15">Hello, World!</h1>

<!-- Loop variant -->
<span class="ease-typewriter-loop ease-chars-20">Type. Erase. Repeat.</span>

<!-- Custom char count via CSS variable -->
<p style="--ease-typewriter-chars: 42"
   class="ease-typewriter">Building with EaseMotion CSS.</p>
