Usage Example

Copy the CSS classes below to use in your projects:

<!-- Add to your HTML head -->
<link rel="stylesheet" href="submissions/examples/scrollbar-styling-tm/style.css">

<!-- Apply the utility class -->
<!-- On a scrollable container -->
<div class="scrollbar-thin scrollbar-primary">
  <!-- Your scrollable content here -->
</div>

<!-- Medium variant -->
<div class="scrollbar-medium scrollbar-success">...</div>

<!-- Thick variant -->
<div class="scrollbar-thick scrollbar-rounded">...</div>
