// tests/a11y.test.js import { describe, it, expect } from 'vitest'; import { JSDOM } from 'jsdom'; import axe from 'axe-core'; const components = { button: '<button class="btn btn-primary">Click me</button>', card: '<div class="card"><h2>Title</h2><p>Content</p></div>', // ... more components }; for (const [name, html] of Object.entries(components)) { it(name + ' has no critical/serious a11y violations', async () => { const { window } = new JSDOM(html, { runScripts: 'dangerously' }); const results = await axe.run(window.document); const violations = results.violations .filter(v => ['critical', 'serious'].includes(v.impact)); expect(violations.length).toBe(0); }); }
