Sidebar Conversion Example

Before (physical)
.ease-sidebar {
  position: fixed;
  left: 0;
  top: 0;
  bottom: 0;
  width: 260px;
  transform: translateX(-100%);
}

.ease-sidebar-right {
  left: auto;
  right: 0;
  transform: translateX(100%);
}
After (logical)
.ease-sidebar {
  position: fixed;
  inset-inline-start: 0;
  inset-block: 0;
  width: 260px;
  transform: translate(-100%, 0);
}

.ease-sidebar-right {
  inset-inline-start: auto;
  inset-inline-end: 0;
  transform: translate(100%, 0);
}
