How to Use

<div class="plasma-modal" id="my-modal" role="dialog">
  <div class="plasma-overlay"></div>
  <div class="plasma-modal-content">
    <div class="plasma-header">
      <h2>Modal Title</h2>
      <button class="plasma-close" onclick="closeModal('my-modal')">
        <span></span>
        <span></span>
      </button>
    </div>
    <div class="plasma-body">
      <!-- Content here -->
    </div>
    <div class="plasma-footer">
      <button class="btn-secondary">Cancel</button>
      <button class="btn-primary">Confirm</button>
    </div>
  </div>
</div>
