Usage

<!-- Add the class to any container -->
<div class="ease-stagger-fade-in">
  <div>Child 1</div>
  <div>Child 2</div>
  <div>Child 3</div>
</div>

<!-- Customize timing via CSS variables -->
<div class="ease-stagger-slide-up"
     style="--ease-stagger-delay: 150ms; --ease-stagger-duration: 0.5s;">
  ...
</div>
