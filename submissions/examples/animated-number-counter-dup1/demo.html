Usage

<style>
@property --num { syntax: "<integer>"; inherits: false; initial-value: 0; }
.c-1000 { animation: count-1000 var(--ease-counter-duration, 3s) ease-out forwards; }
@keyframes count-1000 { from { --num: 0; } to { --num: 1000; } }
</style>
<div class="ease-number-counter c-1000">
  <span class="ease-counter-label">Downloads</span>
</div>
