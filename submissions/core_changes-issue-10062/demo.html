All Values

Both ease-row-gap-* and ease-col-gap-* support values: 0, 1, 2, 3, 4, 6, 8, 10, 12, 16 — matching the --ease-space-* token scale. 

/* Row gap */
.ease-row-gap-0   { row-gap: 0; }
.ease-row-gap-1   { row-gap: 0.25rem; }
.ease-row-gap-2   { row-gap: 0.5rem; }
.ease-row-gap-3   { row-gap: 0.75rem; }
.ease-row-gap-4   { row-gap: 1rem; }
.ease-row-gap-6   { row-gap: 1.5rem; }
.ease-row-gap-8   { row-gap: 2rem; }
.ease-row-gap-10  { row-gap: 2.5rem; }
.ease-row-gap-12  { row-gap: 3rem; }
.ease-row-gap-16  { row-gap: 4rem; }

/* Column gap (same values) */
.ease-col-gap-0   { column-gap: 0; }
.ease-col-gap-1   { column-gap: 0.25rem; }
.ease-col-gap-2   { column-gap: 0.5rem; }
.ease-col-gap-3   { column-gap: 0.75rem; }
.ease-col-gap-4   { column-gap: 1rem; }
.ease-col-gap-6   { column-gap: 1.5rem; }
.ease-col-gap-8   { column-gap: 2rem; }
.ease-col-gap-10  { column-gap: 2.5rem; }
.ease-col-gap-12  { column-gap: 3rem; }
.ease-col-gap-16  { column-gap: 4rem; }

HTML Structure

<!-- Grid with different row/column gaps -->
<div class="ease-grid ease-grid-cols-3 ease-row-gap-8 ease-col-gap-4">
  <div class="ease-card">Card 1</div>
  ...
</div>

<!-- Flex column: row gap only -->
<div class="ease-flex ease-flex-col ease-row-gap-6 ease-col-gap-0">
  <div>Item 1</div>
</div>

<!-- Override uniform gap on one axis -->
<div class="ease-gap-4 ease-row-gap-8">
  <!-- row-gap: 8, column-gap stays 4 -->
</div>
